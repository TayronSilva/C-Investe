DROP DATABASE cinveste;
CREATE DATABASE cinveste;
USE cinveste;
CREATE TABLE usuario (
    id_usuario INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    email VARCHAR(100) NOT NULL UNIQUE,
    telefone VARCHAR(20),
    data_nascimento DATE,
    cep VARCHAR(10),
    estado VARCHAR(2),
    municipio VARCHAR(100),
    rua VARCHAR(150),
    numero VARCHAR(10),
    complemento VARCHAR(100),
    senha VARCHAR(255)
);

CREATE TABLE empreendedor (
    id_empreendedor INT PRIMARY KEY,
    instituicao_ensino VARCHAR(150) NOT NULL,
    curso VARCHAR(100) NOT NULL,
    nivel_ensino VARCHAR(50) NOT NULL,
    ano_expedicao DATE NOT NULL,
    ano_conclusao DATE NOT NULL,
    CONSTRAINT fk_empreendedor_usuario FOREIGN KEY (id_empreendedor) REFERENCES usuario(id_usuario)
);

CREATE TABLE investidor (
    id_investidor INT PRIMARY KEY,
    ocupacao VARCHAR(100) NOT NULL,
    empresa VARCHAR(150) NOT NULL,
    tempo_atuacao INT NOT NULL,
    renda_mensal DECIMAL(12,2),
    valor_investimentos DECIMAL(12,2),
    interesses TEXT,
    CONSTRAINT fk_investidor_usuario FOREIGN KEY (id_investidor) REFERENCES usuario(id_usuario)
);
